                            x<sub>ensemble</sub> = γx<sub>Mt,j</sub> + (1−γ)x<sub>y,i</sub>
                        </div>

                            y<sub>ensemble</sub> = γy<sub>Mt,j</sub> + (1−γ)y<sub>y,i</sub>
                        </div>
                        <div class="formula">
                            W<sub>ensemble</sub> = γW<sub>Mt,j</sub> + (1−γ)W<sub>y,i</sub>
                        </div>
